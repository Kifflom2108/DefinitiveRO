	next;

	switch(select(
		"Poring e Drops",
		"Fabre e Pupa",
		"Lunatic e Chonchon",
		"Pickys",
		"Willow e Savage Babe",
		"Tarou e Roda Frog",
		"Thief Bug e Spore",
		"Thief Bug Famale e Baby Desert Wolf"
	)) {
		case 1:
			warp("novice00",96,48);
			break;
		case 2:
			warp("novice01",96,48);
			break;
		case 3:
			warp("novice02",96,48);
			break;
		case 4:
			warp("novice03",96,48);
			break;
		case 5:
			warp("novice04",96,48);
			break;

			break;

			break;
		case 8:
			warp("novice07",96,48);
			break;
	}
	cutin "", 255; // Remove a imagem ao final do diálogo
	close;
}
