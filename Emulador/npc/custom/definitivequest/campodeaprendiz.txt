prontera,150,150,4	script	PoringQuestNPC	101,{
    
    if (Quest_Poring == 1) { // Verifica se a quest já foi concluída
        mes "Você já completou esta quest.";
        close;
    }

    if (Quest_Poring == 0) { // Se a quest ainda não foi aceita
        mes "Olá, aventureiro!";
        mes "Eu preciso que você mate 10 Porings para me ajudar.";
        mes "Você aceita essa missão?";
        next;
        if (select("Sim:Não") == 1) {
            set Quest_Poring, 2; // Marca a quest como ativa
            set Poring_Count, 0; // Inicializa o contador de Porings
            setquest 61000; // Adiciona a quest (ID da quest configurada no seu `quest_db`)
            questinfo_refresh(); // Atualiza as informações da aba de quests no cliente
            mes "Ótimo! Volte quando tiver matado 10 Porings.";
            close;
        } else {
            mes "Tudo bem, volte quando estiver pronto.";
            close;
        }
    }

    if (Quest_Poring == 2) { // Se a quest está ativa
        if (Poring_Count < 10) { // Ainda não matou os 10 Porings
            mes "Você ainda não completou a missão.";
            mes "Porings derrotados: " + Poring_Count + "/10";
            close;
        } else { // Se completou a missão
            mes "Parabéns! Você completou a missão.";
            mes "Aqui está sua recompensa.";
            getitem 501, 10; // Dá 10 Poções como recompensa
            set Quest_Poring, 1; // Marca a quest como concluída
            erasequest 61000; // Remove a quest da aba de quests
            close;
        }
    }

    close;


// Esse evento é chamado sempre que o jogador mata um monstro
OnNPCKillEvent:
    if (Quest_Poring == 2 && killedrid == 1002) { // ID do Poring = 1002
        set Poring_Count, Poring_Count + 1; // Incrementa o contador de Porings
        if (Poring_Count == 10) {
            dispbottom "Você completou a missão de matar 10 Porings! Volte ao NPC.";
        } else {
            dispbottom "Porings derrotados: " + Poring_Count + "/10.";
        }
        questinfo_refresh(); // Atualiza a aba de quests no cliente
    }
    end;
}
